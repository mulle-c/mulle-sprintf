if( NOT __MULLE_SPRINTF_CMAKE_DEPENDENCIES_TXT__)
   set( __MULLE_SPRINTF_CMAKE_DEPENDENCIES_TXT__ ON)

   message( STATUS "# Include mulle-sprintf CMakeDependencies.txt")

   include( ${DEPENDENCIES_DIR}/include/mulle_buffer/CMakeDependencies.txt)
   include( ${DEPENDENCIES_DIR}/include/mulle_utf/CMakeDependencies.txt)

   if( NOT MULLE_SPRINTF_LIBRARY)
      find_library( MULLE_SPRINTF_LIBRARY NAMES mulle_sprintf)
      message( STATUS "MULLE_SPRINTF_LIBRARY is ${MULLE_SPRINTF_LIBRARY}")
      set( C_DEPENDENCY_LIBRARIES
         ${MULLE_SPRINTF_LIBRARY}
         ${C_DEPENDENCY_LIBRARIES}
         CACHE INTERNAL "need to cache this"
      )
   endif()

   if( MSVC)
      set( MULLE_SPRINTF_DEFINITIONS
         ${MULLE_BUFFER_DEFINITIONS}
         ${MULLE_UTF_DEFINITIONS}
         "-DMULLE_SPRINTF_EXTERN_GLOBAL=extern"
      )
      list( REMOVE_DUPLICATES MULLE_SPRINTF_DEFINITIONS)
   endif()
endif()
